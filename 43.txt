#include <stdio.h>
#include <math.h>
#include <stdlib.h>
int main()
{
	int i, e, p, n, t, check;
	printf("\nNhap N thoa man yeu cau de bai: ");
	scanf("%d", &n);
p:
	check = 0;
	p = 2 + rand() % (999); // Min + Rand % (Max + 1 - Min)
	for (i = 2; i < sqrt(p); i++)
	{
		if (p % i == 0)
			check = 1;
	}
	if (check == 1)
		goto p;

	printf("Gia tri cua p= %d\n", p);
	printf("Nhung so a thoa man de bai la: ");

	int j = 0, a, b, d, k, K[50];
	k = p;
	while (k > 0)
	{
		K[j] = k % 2;
		k = (k - K[j]) / 2;
		j++;
	}
	for (d = 1; d < 100; d++)
	{
		check = 1;
		a = d;

		if (K[0] == 1)
			b = a;

		for (i = 0; i < j; i++)
		{
			a = (a * a) % n;
			if (K[i] == 1)
			{
				b = (a * b) % n;
			}
		}
		if (b == 2)
			printf("%d", d);
		for (i = 2; i < sqrt(b); i++)
		{
			if (b % i == 0)
				check = 0;
		}
		if (check == 1)
			printf("%d ", d);
	}
}
